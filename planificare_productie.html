⚙ Configurare Google Sheets

Conectează aplicația la Google Sheets pentru salvarea datelor
Pași configurare Google Sheets:
  1. Creează un Google Sheet nou și adaugă foile: Istoric, Ambalare_Ref, Procesare_Ref. Foile S26_2026, M26_2026 etc. se creează automat la prima salvare.
  2. În foaia Procesare_Ref copiază datele din sheet-ul PROCESARE din Excel (coloanele: DENUMIRE, Lot OPTIM, BUC OPTIM, LAPTE VACA, LAPTE CAPRA, LAPTE OAIE, LAPTE DEG)
  3. În foaia Ambalare_Ref copiază datele din sheet-ul Ambalare (coloanele: DENUMIRE, Lot OPTIM UM, Timp necesar)
  4. Deschide Extensions → Apps Script și copiază codul de mai jos
  5. Click Deploy → New deployment → Web App, setează acces Anyone
  6. Copiază URL-ul generat și lipește-l mai sus
📋 Cod Apps Script (copiază în Extensions → Apps Script):
const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = e.parameter.sheet;
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify({data:[]}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const data = sheet.getDataRange().getValues();
  return ContentService.createTextOutput(JSON.stringify({data: data}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const req = JSON.parse(e.postData.contents);

    if (req.action === 'saveFull') {
      // 1. Salvează planificarea
      writeSheet(ss, req.planifSheet, req.planifRows);

      // 2. Salvează monitorizarea
      writeSheet(ss, req.monitSheet, req.monitRows);

      // 3. Actualizează Istoric
      const istoricSheet = ss.getSheetByName('Istoric') || ss.insertSheet('Istoric');
      const now = new Date().toLocaleString('ro-RO');
      const existing = istoricSheet.getDataRange().getValues();
      const idx = existing.findIndex(r => r[0] === req.planifSheet);
      if (idx >= 1) {
        istoricSheet.getRange(idx + 1, 1, 1, 3).setValues([[req.planifSheet, req.istoricLabel, now]]);
      } else if (idx === -1 || existing.length <= 1) {
        if (existing.length === 0 || (existing.length === 1 && !existing[0][0])) {
          istoricSheet.clearContents();
          istoricSheet.getRange(1, 1, 1, 3).setValues([['key', 'label', 'savedAt']]);
        }
        istoricSheet.appendRow([req.planifSheet, req.istoricLabel, now]);
      }

      return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // Scriere simplă (monitorizare individuală)
    if (req.action === 'write') {
      writeSheet(ss, req.sheet, req.rows);
      return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService.createTextOutput(JSON.stringify({status:'error', error:'unknown action'}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function writeSheet(ss, name, rows) {
  let sheet = ss.getSheetByName(name);
  if (!sheet) sheet = ss.insertSheet(name);
  sheet.clearContents();
  if (rows && rows.length > 0) {
    sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
  }
}
