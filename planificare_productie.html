⚙ Configurare Google Sheets

Conectează aplicația la Google Sheets pentru salvarea datelor
Pași configurare Google Sheets:
  1. Creează un Google Sheet nou și adaugă 4 foi: Planificare, Monitorizare, Ambalare_Ref, Procesare_Ref
  2. În foaia Procesare_Ref copiază datele din sheet-ul PROCESARE din Excel (coloanele: DENUMIRE, Lot OPTIM, BUC OPTIM, LAPTE VACA, LAPTE CAPRA, LAPTE OAIE, LAPTE DEG)
  3. În foaia Ambalare_Ref copiază datele din sheet-ul Ambalare (coloanele: DENUMIRE, Lot OPTIM UM, Timp necesar)
  4. Deschide Extensions → Apps Script și copiază codul de mai jos
  5. Click Deploy → New deployment → Web App, setează acces Anyone
  6. Copiază URL-ul generat și lipește-l mai sus
📋 Cod Apps Script (copiază în Extensions → Apps Script):
const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(data.sheet);
  
  if (data.action === 'write') {
    const rows = data.rows;
    sheet.clearContents();
    if (rows.length > 0) {
      sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({status:'error'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = e.parameter.sheet;
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify({error:'Sheet not found'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const data = sheet.getDataRange().getValues();
  return ContentService.createTextOutput(JSON.stringify({data: data}))
    .setMimeType(ContentService.MimeType.JSON);
}
